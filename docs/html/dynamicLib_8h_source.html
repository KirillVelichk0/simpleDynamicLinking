225 [[nodiscard]] auto CallFuncFromNameSafeInternal(const std::string &funcName)
226 {
227 ResultType* result;
228 std::string fullFuncName = funcName + "_ExternC_ViaDynamicLib";
229 if (!this->isActive())
230 {
231 throw std::runtime_error("Controller is not active");
232 }
233 std::any *preResult;
234 void *rawFuncPointer = this->GetRawFuncCaller(fullFuncName);
235 if (rawFuncPointer == nullptr)
236 {
237 throw std::runtime_error("Function with name "s + funcName + " does not imported");
238 }
239 auto creater = (void *(*)())(rawFuncPointer);
240 void *rawImported = creater();
241 if (rawImported == nullptr)
242 {
243 throw std::runtime_error("Function with name "s + funcName + " return nullptr pointer");
244 }
245 preResult = reinterpret_cast<std::any *>(rawImported);
246 if (!preResult->has_value())
247 {
248 throw std::runtime_error("Something is wrong with imported pointer");
249 }
250 try
251 {
252 auto packedPointer = std::any_cast<std::shared_ptr<PackedPointer<ResultType *>>>(*preResult);
253 result = packedPointer->Unpack();
254 }
255 catch (std::bad_any_cast &_)
256 {
257 try{
258 auto err = std::any_cast<ImportError>(*preResult);
259 delete preResult;
260 throw std::runtime_error(err.errorStr);
261 } catch(std::bad_any_cast& _){
262 delete preResult;
263 throw std::runtime_error("Uncorrect result type");
264 }
265
266 }
267 delete preResult;
268 return result;
269 }
